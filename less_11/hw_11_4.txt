При помощи базы данных Redis решите задачу поиска имени пользователя по 
электронному адресу и наоборот, поиск электронного адреса пользователя по его имени.

server@server:~/lr_sql/prg/lr_sql_1$ redis-cli 
127.0.0.1:6379> HSET user_emails "1" "111@111.com"
(integer) 1
127.0.0.1:6379> HSET user_names "1" "11111"
(integer) 1
127.0.0.1:6379> HSET user_emails "2" "222@222.com"
(integer) 1
127.0.0.1:6379> HSET user_names "2" "22222"
(integer) 1
127.0.0.1:6379> HGET user_names "2"
"22222"
127.0.0.1:6379> HGET user_emails "2"
"222@222.com"
127.0.0.1:6379> HGET user_emails "1"
"111@111.com"
127.0.0.1:6379> HGET user_names "1"
"11111"
127.0.0.1:6379> exit
server@server:~/lr_sql/prg/lr_sql_1$
